xs! {
    package Array::Util;

    sub sum_array(ctx) {
        let param: SV = ctx.st_fetch(0);
        let array: AV = param.deref_av().expect("not an array ref");
        let mut sum = 0.0;
        for index in 0..array.top_index() + 1 {
            if let Some(item) = array.fetch(index) {
                sum += item;
            }
        }
        xs_return!(ctx, sum);
    }
}
